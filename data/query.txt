SELECT
    id,
    stock_id,
    sku,
    product_name,
    SUM(quantity) AS total_quantity
FROM
    orders
WHERE
    created_at BETWEEN '2025-07-01' AND '2025-08-31'
GROUP BY
    sku,
    product_name
ORDER BY
    `orders`.`stock_id` ASC


SELECT
    oi.*
FROM
    order_invoices oi
JOIN sale_points sp ON
    sp.id = oi.sale_point_id
WHERE
    sp.code_number = 'WGP273';
